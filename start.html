import openai import tkinter as tk from tkinter import scrolledtext import pyttsx3 import speech_recognition as sr from threading import Thread # ========== SETUP ========== # Yahan apni real key daalein client = openai.OpenAI(api_key="YOUR_API_KEY") engine = pyttsx3.init() recognizer = sr.Recognizer() # ========== Functions ========== def speak(text): engine.say(text) engine.runAndWait() def listen(): try: with sr.Microphone() as source: chat_box.insert(tk.END, "Listening...\n") chat_box.see(tk.END) audio = recognizer.listen(source, timeout=5) return recognizer.recognize_google(audio) except: return "" def chat_with_ai(prompt): # Naya OpenAI format response = client.chat.completions.create( model="gpt-3.5-turbo", messages=[ {"role": "system", "content": "You are a helpful AI assistant."}, {"role": "user", "content": prompt} ], max_tokens=150 ) return response.choices[0].message.content def ai_response(user_input): answer = chat_with_ai(user_input) chat_box.insert(tk.END, f"AI: {answer}\n\n") chat_box.see(tk.END) speak(answer) with open("chat_history.txt", "a", encoding="utf-8") as f: f.write(f"You: {user_input}\nAI: {answer}\n\n") def send_text(): user_input = text_input.get() if not user_input.strip(): return text_input.delete(0, tk.END) chat_box.insert(tk.END, f"You: {user_input}\n") Thread(target=ai_response, args=(user_input,)).start() def send_voice(): def voice_thread(): user_input = listen() if not user_input.strip(): chat_box.insert(tk.END, "Voice not recognized.\n") return chat_box.insert(tk.END, f"You (voice): {user_input}\n") ai_response(user_input) Thread(target=voice_thread).start() # ========== GUI ========== root = tk.Tk() root.title("My AI Chatbot") root.geometry("600x550") chat_box = scrolledtext.ScrolledText(root, wrap=tk.WORD, width=70, height=25) chat_box.pack(padx=10, pady=10) text_input = tk.Entry(root, width=50) text_input.pack(side=tk.LEFT, padx=10) tk.Button(root, text="Send", command=send_text).pack(side=tk.LEFT) tk.Button(root, text="🎤 Voice", command=send_voice).pack(side=tk.LEFT) root.mainloop()
